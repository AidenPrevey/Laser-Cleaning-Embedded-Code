97std::complex<double> inline s2z(std::complex<double> s, double Ts)
98{
99 return (static_cast<std::complex<double>>(1.0) + (Ts / 2) * s) /
100 (static_cast<std::complex<double>>(1.0) - (Ts / 2) * s);
101}

115template <uint8_t ORDER>
116std::array<double, ORDER + 1> expandPolynomial(std::array<std::complex<double>, ORDER> zeros)
117{
118 std::array<std::complex<double>, ORDER + 1> coefficients{
119 std::complex<double>(1.0, 0.0)}; // Start with 1
120
121 for (size_t i = 0; i < ORDER; i++)
122 {
123 // Multiply current polynomial by (x - zero)
124 std::array<std::complex<double>, ORDER + 1> newCoefficients{std::complex<double>(0.0, 0.0)};
125 for (size_t j = 0; j < i + 1; j++)
126 {
127 newCoefficients[j] -=
128 coefficients[j] * zeros[i]; // Multiply current coefficient by zero
129 newCoefficients[j + 1] += coefficients[j]; // Shift current coefficient
130 }
131 coefficients.swap(newCoefficients);
132 }
133 std::array<double, ORDER + 1> stripped_coefficients{};
134 // Extract the real part of each coefficient, the imaginary appears to be an
135 // artifact of double
136 for (size_t i = 0; i < ORDER + 1; i++)
137 {
138 stripped_coefficients[i] = coefficients[i].real();
139 }
140
141 return stripped_coefficients;
142}

154template <uint8_t ORDER>
155std::complex<double> evaluateFrequencyResponse(
156 const std::array<double, ORDER + 1> &b,
157 const std::array<double, ORDER + 1> &a,
158 double w,
159 double Ts)
160{
161 std::complex<double> j(0, 1); // imaginary unit i = sqrt(-1)
162 std::complex<double> numerator(0, 0);
163 std::complex<double> denominator(0, 0);
164
165 double omega = w * Ts; // omega is in terms of rad/sample
166
167 // Evaluate numerator: b0 + b1 * e^{-jω} + b2 * e^{-j2ω} + ...
168 for (size_t k = 0; k < b.size(); ++k)
169 {
170 numerator += b[k] * (std::cos(omega * static_cast<double>(k)) -
171 j * std::sin(omega * static_cast<double>(k)));
172 }
173
174 // Evaluate denominator: a0 + a1 * e^{-jω} + a2 * e^{-j2ω} + ...
175 for (size_t k = 0; k < a.size(); ++k)
176 {
177 denominator += a[k] * (std::cos(omega * static_cast<double>(k)) -
178 j * std::sin(omega * static_cast<double>(k)));
179 }
180
181 return numerator / denominator;
182}

199std::complex<double> inline complexSqrt(std::complex<double> z)
200{
201 double r = std::sqrt(complexAbs(z));
202 double theta = static_cast<double>(std::atan2(z.imag(), z.real())) * static_cast<double>(0.5f);
203 return {r * std::cos(theta), r * std::sin(theta)};

219template <uint8_t ORDER, FilterType Type = LOWPASS, typename T = float>
220Coefficients<getNumCoefficients(ORDER, Type), T> butterworth(double wc, double Ts, double wh = 0.0)
221{
222 const uint16_t COEFFICIENTS = getNumCoefficients(ORDER, Type);
223
224 std::array<T, COEFFICIENTS> naturalResponseCoefficients;
225 std::array<T, COEFFICIENTS> forcedResponseCoefficients;
226
227 const int n = ORDER;
228
229 // For band filters we treat wc as ωl
230 double wl = wc;
231 double whp = wh;
232 std::array<std::complex<double>, 2 * ORDER> bandpass_stop_poles;
233
234 // pre-warp all edges for bilinear transform
235 wl = static_cast<double>(2.0) / Ts * std::tan(wl * (Ts / static_cast<double>(2.0)));
236 whp = static_cast<double>(2.0) / Ts * std::tan(whp * (Ts / static_cast<double>(2.0)));
237
238 // generate N prototype poles on unit circle
239 std::array<std::complex<double>, COEFFICIENTS - 1> poles;
240 for (int k = 0; k < n; ++k)
241 {
242 double theta = M_PI * (2.0 * k + 1) / (2.0 * n) + M_PI / 2.0;
243 poles[k] = std::complex<double>(std::cos(theta), std::sin(theta));
244 }
245
246 std::array<std::complex<double>, COEFFICIENTS - 1> zPoles;
247
248 // apply the appropriate s-domaisn transform to each pole
249 switch (Type)
250 {
251 case LOWPASS:
252 {
253 // poles are multiplied by wl to scale them to the desired cutoff frequency
254 for (int j = 0; j < n; ++j) poles[j] = poles[j] * wl;
255
256 // now map each analog pole into the z-plane
257 for (int i = 0; i < COEFFICIENTS - 1; ++i) zPoles[i] = s2z(poles[i], Ts);
258 break;
259 }
260 case HIGHPASS:
261 {
262 // applys the inverse transform of the butterworth lowpass filter
263 for (int j = 0; j < n; ++j)
264 {
265 poles[j] = wl / poles[j];
266 }
267 // now map each analog pole into the z-plane
268 for (int i = 0; i < COEFFICIENTS - 1; ++i) zPoles[i] = s2z(poles[i], Ts);
269 break;
270 }
271 // In the case of a bandpass or a bandstop the amount of poles doubles.
272 case BANDPASS:
273 {
274 /*
275 * transform in the form of s → (s² + Ω₀²) / (B · s)
276 * where:
277 * Ω₀ = √(Ω_low * Ω_high) // Center frequency (rad/sec)
278 * B = Ω_high - Ω_low // Bandwidth (rad/sec)
279 */
280 double B = whp - wl;
281 double W0sq = whp * wl;
282
283 for (int j = 0; j < ORDER; ++j)
284 {
285 std::complex<double> p = poles[j];
286
287 std::complex<double> discriminant =
288 (p * B) * (p * B) - static_cast<std::complex<double>>(4.0) * W0sq;
289 std::complex<double> root = complexSqrt(discriminant);
290
291 bandpass_stop_poles[2 * j] =
292 (p * B + root) * static_cast<std::complex<double>>(0.5);
293 bandpass_stop_poles[2 * j + 1] =
294 (p * B - root) * static_cast<std::complex<double>>(0.5);
295 }
296
297 // now map each analog pole into the z-plane
298 for (int i = 0; i < COEFFICIENTS - 1; ++i) zPoles[i] = s2z(bandpass_stop_poles[i], Ts);
299
300 break;
301 }
302
303 case BANDSTOP:
304 {
305 /*
306 * transform in the form of s → B · s / (s² + Ω₀²)
307 * where:
308 * Ω₀ = √(Ω_low * Ω_high) // Center frequency (rad/sec)
309 * B = Ω_high - Ω_low // Bandwidth (rad/sec)
310 */
311 double B = whp - wl;
312 double W0sq = whp * wl;
313 for (int j = 0; j < n; ++j)
314 {
315 std::complex<double> p = poles[j];
316
317 std::complex<double> discriminant =
318 B * B - (static_cast<std::complex<double>>(4.0) * -p * W0sq);
319 std::complex<double> root = complexSqrt(discriminant);
320
321 bandpass_stop_poles[2 * j] =
322 (B + root) / (static_cast<std::complex<double>>(2.0) * p);
323 bandpass_stop_poles[2 * j + 1] =
324 (B - root) / (static_cast<std::complex<double>>(2.0) * p);
325 }
326
327 // now map each analog pole into the z-plane
328 for (int i = 0; i < COEFFICIENTS - 1; ++i) zPoles[i] = s2z(bandpass_stop_poles[i], Ts);
329
330 break;
331 }
332 }
333
334 std::array<std::complex<double>, COEFFICIENTS - 1> zZeros;
335
336 switch (Type)
337 {
338 case LOWPASS:
339 // zeros: for Butterworth lowpass all z-zeros at z = –1
340 zZeros.fill(std::complex<double>(-1, 0));
341 break;
342 case HIGHPASS:
343 // zeros: for butterworth highpass all z-zeros are at z = 1
344 zZeros.fill(std::complex<double>(1, 0));
345 break;
346 case BANDPASS:
347 // zeros: for butterworth bandpass all z-zeros are at z = ±1
348 for (int i = 0; i < COEFFICIENTS - 1; ++i)
349 {
350 zZeros[i] = (i % 2 == 0) ? std::complex<double>(1, 0) : std::complex<double>(-1, 0);
351 }
352 break;
353 case BANDSTOP:
354 {
355 /* zeros: for butterworth bandstop are in a circle with radius 1 at the
356 * center of the z-domain with an angle in radians per sample of the
357 * center frequency. The zeros are complex conjugates of each other.
358 */
359
360 /* the notch (center) frequency in radians/sample */
361 double omega0 = std::sqrt(wl * whp) * Ts;
362
363 double realPart = std::cos(omega0);
364 double imagPart = std::sin(omega0);
365
366 std::complex<double> zeroPlus(realPart, imagPart); // e^(+jω0)
367 std::complex<double> zeroMinus(realPart, -imagPart); // e^(-jω0)
368
369 for (int i = 0; i < COEFFICIENTS - 1; i += 2)
370 {
371 zZeros[i] = zeroPlus; // first of pair
372 if (i + 1 < COEFFICIENTS) zZeros[i + 1] = zeroMinus; // second of pair
373 }
374 }
375 break;
376 }
377
378 // get expanded polynomials
379 auto b = expandPolynomial<COEFFICIENTS - 1>(zZeros);
380 auto a = expandPolynomial<COEFFICIENTS - 1>(zPoles);
381
382 // scale the gain properly
383 switch (Type)
384 {
385 case LOWPASS:
386 {
387 // Eval at DC
388 auto freqResp = evaluateFrequencyResponse<COEFFICIENTS - 1>(b, a, 0, Ts);
389 auto mag = complexAbs(freqResp);
390 double scale = 1 / mag;
391 for (auto &coef : b) coef *= scale;
392 break;
393 }
394 case HIGHPASS:
395 {
396 // Eval at nyquist
397 auto freqResp = evaluateFrequencyResponse<COEFFICIENTS - 1>(
398 b,
399 a,
400 static_cast<double>(M_PI) / Ts,
401 Ts);
402 auto mag = complexAbs(freqResp);
403 double scale = 1 / mag;
404 for (auto &coef : b) coef *= scale;
405 break;
406 }
407 case BANDPASS:
408 {
409 // Eval at center frequency
410 auto freqResp =
411 evaluateFrequencyResponse<COEFFICIENTS - 1>(b, a, std::sqrt(wl * whp), Ts);
412 auto mag = complexAbs(freqResp);
413 double scale = 1 / mag;
414 for (auto &coef : b) coef *= scale;
415 break;
416 }
417 case BANDSTOP:
418 {
419 // Eval at dc gain
420 auto freqResp = evaluateFrequencyResponse<COEFFICIENTS - 1>(b, a, 0, Ts);
421 auto mag = complexAbs(freqResp);
422 double scale = 1 / mag;
423 for (auto &coef : b) coef *= scale;
424 break;
425 }
426 }
427
428 // store in member arrays (reverse order)
429 for (size_t i = 0; i < COEFFICIENTS; ++i)
430 {
431 naturalResponseCoefficients[COEFFICIENTS - i - 1] = a[i];
432 forcedResponseCoefficients[COEFFICIENTS - i - 1] = b[i];
433 }
434
435 // Store in
436 Coefficients<COEFFICIENTS, T> both_coefficients;
437 both_coefficients.naturalResponseCoefficients = naturalResponseCoefficients;
438 both_coefficients.forcedResponseCoefficients = forcedResponseCoefficients;
439 return both_coefficients;
440}
